k_wendia = {
	color = { 175 50 70 }
	
	capital = c_brandenburg
	
	can_create = {
		trigger_if = {
			limit = {
				is_ai = yes
			}
			NOT = {
				has_title = title:e_hre
			}
		}
		custom_tooltip = {
			text = hre_cannot_create_kingdom
			is_not_hre_vassal = {
				CHARACTER = root
			}
		}
		trigger_if = {
			limit = {
				is_ai = yes
			}
			OR = {
				culture = culture:polabian
				culture = culture:obotrite
				culture = culture:ranian
				culture = culture:lutician
				culture = culture:sorbian
			}
		}
	}
}
